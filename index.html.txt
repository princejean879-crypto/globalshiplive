<script>
    const packages = {
        "DEMO001": {status:"Delivered",statusClass:"delivered",location:"Paris, France",lastUpdate:"2025-05-16",eta:"May 16, 2025",history:["May 14 - Picked up","May 15 - In transit Lyon","May 16 - Delivered"]},
        "DEMO002": {status:"In Transit",statusClass:"transit",location:"Lyon, France",lastUpdate:"2025-05-17",eta:"May 19, 2025",history:["May 15 - Label created","May 16 - Picked up","May 17 - In transit Lyon"]}
    };
    document.getElementById('trackBtn').onclick = () => {
        const num = document.getElementById('trackingInput').value.trim();
        const msg = document.getElementById('messageArea');
        const result = document.getElementById('resultArea');
        if(!num){msg.innerHTML='<p style="color:red">Enter a number</p>';result.innerHTML='';return;}
        msg.innerHTML='<p>🔍 Searching...</p>';
        setTimeout(()=>{
            const data = packages[num];
            if(data){
                msg.innerHTML='<p style="color:green">✅ Package found</p>';
                let history = '<div class="timeline">';
                data.history.forEach(h=>{history+=`<div class="timeline-item">📌 ${h}</div>`;});
                history+='</div>';
                result.innerHTML = <div class="result"><div class="${data.statusClass}">${data.status}</div><p>📍 ${data.location}</p><p>📅 ${data.eta}</p>${history}</div>;
            } else {
                msg.innerHTML='<p style="color:red">❌ Not found. Try: DEMO001 or DEMO002</p>';
                result.innerHTML='';
            }
        },800);
    };
</script>
